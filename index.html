  <script>

    async function uploadFile() {

      const fileInput = document.getElementById('fileInput');

      const status = document.getElementById('status');

      const download = document.getElementById('download');


      if (!fileInput.files.length) {

        status.textContent = "請選擇影片檔案。";

        return;

      }


      const file = fileInput.files[0];

      const formData = new FormData();

      formData.append("file", file);


      status.textContent = "正在上傳與轉換中，請稍候... ⏳";


      try {

        const res = await fetch("https://你的-n8n-domain/webhook/upload-subtitle", {

          method: "POST",

          body: formData

        });


        const data = await res.json();


        if (data.download_url) {

          status.textContent = "處理完成 ✅";

          download.innerHTML = `<a href="${data.download_url}" target="_blank">🎬 點我下載字幕影片</a>`;

        } else {

          status.textContent = "處理失敗，請稍後再試 ❌";

        }

      } catch (e) {

        status.textContent = "發生錯誤：「" + e.message + "」";

      }

    }

  </script>
